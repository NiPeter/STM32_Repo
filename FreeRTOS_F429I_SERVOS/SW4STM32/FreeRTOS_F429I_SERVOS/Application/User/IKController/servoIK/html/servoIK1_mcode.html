1function [ alpha ] = servoIK( Q ,ik )
 2%SERVOIK
 3
 4% Patrz -> Dysk Ti -> Steward_servo.pdf
 5% + Szufnarowski Steward Platform - low cost study
 6
 7%Błąd w opracowaniu Steward_servo! Wektor Pi powinien być wyrażony w
 8%układzie Bazy, a nie, jak jak zapisano w Steward_servo.pdf, w układzie
 9%platformy.
10
11%Niezbędna zmiana w parametrze B względem Szufnarowskiego, w opracowaniu
12%zakładają że serwa zwrócone są do środka platformy! Tutaj, zwrócone na
13%zewnątrz.
14
15%% Zmienne statyczne
16persistent prev_alpha
17persistent prev_Q
18
19if isempty(prev_Q)
20    prev_Q = [0 0 0 0 0 0];
21end
22if isempty(prev_alpha)
23    prev_alpha = [0 3.1416 0 3.1416 0 3.1416];
24end
25
26
27%global ik
28%% Tworzenie macierzy obrotu i wektora pozycji
29
30[T,R] = q2tr(Q);
31T(3) = T(3) + ik.h0;
32
33%% Lokalne przypisanie parametrów platformy
34
35s = ik.servo.s; % długoś popychacza
36a = ik.servo.a; % długość orczyka
37beta = ik.servo.beta; % wektor kątów mocowania przegubów bazy (silników) od
38% osi X
39p = ik.p; % tablica wektorów pozycji przegubów platformy w układzie platformy
40b = ik.b; % tablica wektorów pozycji przegubów bazy w układzie bazy
41
42%% Obliczenia
43
44alpha = [0 0 0 0 0 0];    %Alokacja pamięci
45
46for i=1:6 % Dla każdego z sześciu serw
47    
48    pb = T+R*p(i,:).'; % wektor pozycji przegubu i-tego platformy w układzie bazy
49    l = norm( pb - b(i,:).'); %długość nogi i-tej
50    
51    %%% Parametry pomocnicze - patrz Szufnarowski
52    A = 2*a*( pb(3) - b(i,3));
53    B = -2*a*(sin(beta(i))*(pb(1) - b(i,1)) - cos(beta(i))*(pb(2)-b(i,2)));
54    C = l^2 - s^2 + a^2;
55    
56    
57    argum = C/sqrt(A^2+B^2); %argument arcsinusa
58    
59    %%% Jeżeli argument arcsinusa jest poza zakresem <-1,1> to brak
60    %%% rozwiązania
61    if ((argum > 1) || (argum<-1))
62        % Tutaj obsłużyć przypadek gdy nie ma rozwiązania
63        %warning('Complex servoIK solution found!\nInput vector \tQ=[%.3f %.3f %.3f %.3f %.3f %.3f]\nUsing \t\t\tQ=[%.3f %.3f %.3f %.3f %.3f %.3f] instead.',Q,prev_Q);
64        alpha =  prev_alpha;
65        Q_out = prev_Q;
66        flag = 0;
67        return
68    else
69        
70        %%% Obliczanie kątów dla silników parzystych i nieparzystych
71        if( mod(i,2) == 1 ) % Silnik nieparzysty - odd, i = 1 3 5
72            alpha(i) =  ( asin(argum)-(atan(B/A)) );
73
74            
75        else                % Silnik parzysty - even, i = 2 4 6
76            alpha(i) = pi + ( asin(-argum)-(atan(B/A)) ) ;
77
78        end
79        
80
81    end
82        
83end
84
85        prev_alpha = alpha;
86        prev_Q = Q;
87        Q_out=Q;
88        flag = 1;
89
90end
