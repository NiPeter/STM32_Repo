1function [ ik ] = structInitFnc( steward )
 2%% Base Init
 3%%% Obliczanie kątów mocowania silników
 4
 5% base = steward.base;
 6% platform = steward.platform;
 7% drive = steward.drive;
 8base_q = [ 0 0 0 0 0 0];
 9platform_q = [ 0 0 0 0 0 0];
10base_q(1) = steward.base.alpha/2;
11for i=2:6
12    if(mod(i,2)==1) % silnik nieparzysty
13        base_q(i) = base_q(i-1)+ steward.base.alpha;
14        
15    else % silnik parzysty
16        base_q(i) = base_q(i-1) - steward.base.alpha + 2/3*pi;
17        
18    end
19end
20
21
22%% Platform Init
23%%% Obliczanie kątów mocowania popychaczy
24
25platform_q(1) = steward.platform.alpha/2;
26for i=2:6
27    if(mod(i,2)==1) % silnik nieparzysty
28        platform_q(i) = platform_q(i-1) + steward.platform.alpha;
29        
30    else % silnik parzysty
31        platform_q(i) = platform_q(i-1) - steward.platform.alpha+ 2/3*pi;
32        
33    end
34end
35
36
37%% IK parameters
38
39%%% alokacja pamięci
40%ik.b = cell(6,1); % tablica wektorów pozycji silników w układzie bazy
41%ik.p = cell(6,1); % tablica wektorów pozycji mocowania popychaczy w
42%układzie platformy
43
44
45ik = struct(...
46'b', zeros(6,3),...
47'p', zeros(6,3), ...
48'h0', [0], ...
49'servo' , struct('a',0,'s',0,'beta',[0 0 0 0 0 0]) ...
50);
51
52%%% Obliczenia
53for i=1:6 % Dla każdego z sześciu serw
54    %%%Obliczanie pozycji i-tego silnika w układzie bazy
55    ik.b(i,1) = steward.base.r * cos(base_q(i));
56    ik.b(i,2) = steward.base.r * sin(base_q(i));
57    ik.b(i,3) = 0;
58    
59    %ik.b{i} = [xb yb 0]'; % wpisanie wektora pozycji silnika do tablicy
60    
61    %%%Obliczanie pozycji mocowania i-tego popychacza w układzie platformy
62    ik.p(i,1) = steward.platform.r * cos(platform_q(i));
63    ik.p(i,2) = steward.platform.r * sin(platform_q(i));
64    ik.p(i,3) = 0;
65    %ik.p{i} = [xp yp 0]'; % wpisanie wektora pozycji popychacza do tablicy
66   
67        %%% Obliczanie wysokości platformy dla alpha = 0
68    if i==3
69        A = rotz(base_q(i)+pi/2)*[steward.drive.a 0 0].';
70        h0 = sqrt( steward.drive.s^2 - (ik.p(i,1)-ik.b(i,1)-A(1))^2 - (ik.p(i,2)-ik.b(i,2)-A(2))^2);
71        ik.h0 = h0;
72    end
73end
74
75
76
77
78%%% Wpisanie parametrów
79%ik.h0 = platform.h; % wysokość zerowa
80ik.servo.a = steward.drive.a; % długość orczyka
81ik.servo.s = steward.drive.s; % długość popychacza
82ik.servo.beta = base_q; % kąty między poszczególnymi silnikami a osią X
83
84end
