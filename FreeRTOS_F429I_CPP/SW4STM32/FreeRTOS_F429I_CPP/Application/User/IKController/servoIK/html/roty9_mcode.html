31function R = roty(t, deg)
32    if nargin > 1 && strcmp(deg, 'deg')
33        t = t *pi/180;
34    end
35    ct = cos(t);
36    st = sin(t);
37    R = [
38        ct  0   st
39        0   1   0
40       -st  0   ct
41       ];
